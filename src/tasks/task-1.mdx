## Terraform Code

```hcl
provider "aws" {
    region     = "ap-south-1"
    access_key = "YOUR_ACCESS_KEY"
    secret_key = "YOUR_SECRET_KEY"
}

resource "aws_instance" "first-server" {
    ami           = "ami-0d03cb826412c6b0f"
    instance_type = "t2.micro"
    tags = {
        Name = "amazon_linux"
    }
}
```
